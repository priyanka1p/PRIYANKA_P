Modify the temperature conversion program to print a heading above the table.
#include <stdio.h>

int main() {
    int fahrenheit;
    printf("    F     C\n");
    printf("------------\n");
    for (fahrenheit = 0; fahrenheit <= 300; fahrenheit+= 20) {
        printf("%5d %5.1f\n", fahrenheit, (5.0/9.0) * (fahrenheit - 32.0));
    }

    return 0;
}
